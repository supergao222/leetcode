class Solution {
    public int romanToInt(String s) {
      int result = 0;
        int val = 0, lastVal = 0;
        for (int i = 0; i != s.length(); i++) {
            switch (s.charAt(i)) {
                case 'I': val = 1; break;
                case 'V': val = 5; break;
                case 'X': val = 10; break;
                case 'L': val = 50; break;
                case 'C': val = 100; break;
                case 'D': val = 500; break;
                case 'M': val = 1000; break;
            }
            switch (val - lastVal) {
                case 4:
                case 40:
                case 400:
                case 9:
                case 90:
                case 900:
                    result += val - 2 * lastVal;
                    break;
                default:
                    result += val;
            }
            lastVal = val;
        }
        return result;  
    }
}
