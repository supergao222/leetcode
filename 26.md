class Solution {
    public int removeDuplicates(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        int cur = nums[0];
        int len = 1;
        for (int i = 0; i != nums.length; i++) {
            if (nums[i] != cur) {
                nums[len++] = nums[i];
                cur = nums[i];
            }
        }
        return len; 
    }
}
