/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        if (head == null) {
            return head;
        }
        int len = 1;
        ListNode end = head;
        while (end.next != null) {
            len++;
            end = end.next;
        }
        if (k % len == 0) {
            return head;
        }
        int step = len - (k % len);
        ListNode iter = head;
        while (--step > 0) {
            iter = iter.next;
        }
        end.next = head;
        head = iter.next;
        iter.next = null;
        return head;
    }
}
