class Solution {
    public int minSwapsCouples(int[] row) {
        int step = 0;
        for (int i = 0; i != row.length; i+=2) {
            int mate = (row[i] % 2 == 0 ? row[i] + 1 : row[i] - 1);
            int matePos = i + 1;
            for (;matePos != row.length; matePos++) {
                if (row[matePos] == mate) {
                    break;
                }
            }
            if (matePos == i + 1) {
                continue;
            }
            int temp = row[i + 1];
            row[i + 1] = row[matePos];
            row[matePos] = temp;
            step++;
        }
        return step;
    }
}
