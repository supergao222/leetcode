class Solution {
    public int threeSumClosest(int[] nums, int target) {
        Arrays.sort(nums);
        int minDelta = Integer.MAX_VALUE;
        int result = 0;
        for (int i = 0; i < nums.length - 2; i++) {
            int l = i + 1, r = nums.length - 1;
            while (l < r) {
                int sum = nums[i] + nums[l] + nums[r];
                int delta = Math.abs(sum - target);
                if (delta < minDelta) {
                    minDelta = delta;
                    result = sum;
                }
                if (sum - target < 0) {
                    l++;
                } else {
                    r--;
                }
            }
        }
        return result;
    }
}
